# COMANDOS PARA EXECUTAR NO VPS (212.85.11.238)

# 1. Verificar processos ativos
ps aux | grep -i whatsapp
ps aux | grep -i node
ps aux | grep -i 3000
ps aux | grep -i 3001

# 2. Verificar portas em uso
netstat -tlnp | grep :3000
netstat -tlnp | grep :3001

# 3. Verificar PM2
pm2 list
pm2 status

# 4. Verificar diretório do serviço
ls -la /var/whatsapp-api/
cd /var/whatsapp-api
ls -la

# 5. Verificar se há arquivo package.json
cat package.json

# 6. Se não houver processo PM2, iniciar o serviço
# Opção 1: Iniciar com PM2
pm2 start index.js --name whatsapp-multi-session
pm2 save
pm2 startup

# Opção 2: Se não houver index.js, procurar outros arquivos
ls -la *.js
# Se encontrar app.js ou server.js:
pm2 start app.js --name whatsapp-multi-session
# ou
pm2 start server.js --name whatsapp-multi-session

# 7. Verificar se o serviço está rodando
pm2 list
pm2 logs whatsapp-multi-session

# 8. Testar conectividade
curl http://localhost:3000/status
curl http://localhost:3001/status

# 9. Se ainda não funcionar, verificar recursos
top
free -h
df -h

# 10. Verificar logs do sistema
journalctl -u whatsapp* --no-pager -n 20
tail -f /var/log/whatsapp*.log 2>/dev/null
